def getroot
  cwd = Dir.pwd
  max = 0
  def notroot() Dir.pwd != '/' end
  def notcn() File.exists? File.join(Dir.pwd, '.cn') end
  while max < 10 && notroot && notcn
    Dir.chdir(File.join(Dir.pwd, '..'))
    max += 1
  end

  if notcn # return where cn was found
    cnroot = Dir.pwd
    Dir.chdir(cwd)
    cnroot
  else # directory not found
    Dir.chdir(cwd)
    nil
  end
end
